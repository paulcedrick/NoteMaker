let NotesService = require('../services/notes');
 
 
module.exports = {
  createNote: function (req, res) {
    let title = req.body.title,
      body = req.body.body,
      params = {
        title,
        body
      };
 
    NotesService.create(params)
    .then(note => {
      res.status(201).json(note.data);
    })
    .catch(err => {
      res.status(err.errorNo).json(err);
    });
  },
  updateNote: (req, res) => {
    let id = req.body.id;
    let title = req.body.title;
    let body = req.body.body;
    let params = {
      id,
      title,
      body
    };
 
    NotesService.update(params, (success, response) => {
      if (success) {
        res.status(201).json(response.data);
      }
      else {
        res.status(500).json({data: response.data});
      }
    });
  },
  deleteNote: (req, res) => {
    let id = req.body.id;
    let params = {
      id: id
    };
    NotesService.delete(params, (success, response) => {
      if (success) {
        res.status(201).json(response.data);
      }
      else {
        res.status(500).json({data: response.data});
      }
    });
  }
};
